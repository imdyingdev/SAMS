import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, Alert } from 'react-native';
import * as MailComposer from 'expo-mail-composer';

export default function EmailConfirmation() {
  const [email, setEmail] = useState('');
  const [code, setCode] = useState('');

  // Generate a random 6-digit confirmation code
  const generateCode = () => {
    return Math.floor(100000 + Math.random() * 900000).toString();
  };

  // Check if mail composer is available
  const checkAvailability = async () => {
    const isAvailable = await MailComposer.isAvailableAsync();
    return isAvailable;
  };

  // Send confirmation email
  const sendConfirmationEmail = async () => {
    if (!email) {
      Alert.alert('Error', 'Please enter an email address');
      return;
    }

    // Check if mail composer is available
    const available = await checkAvailability();
    if (!available) {
      Alert.alert('Error', 'Email service is not available on this device');
      return;
    }

    // Generate confirmation code
    const confirmationCode = generateCode();
    setCode(confirmationCode);

    // Compose email
    try {
      const result = await MailComposer.composeAsync({
        recipients: [email],
        subject: 'Your Confirmation Code',
        body: `
Hello!

Your confirmation code is: ${confirmationCode}

This code will expire in 10 minutes.

If you didn't request this code, please ignore this email.

Thank you!
        `,
        isHtml: false,
      });

      // Handle result
      if (result.status === 'sent') {
        Alert.alert('Success', 'Confirmation email sent!');
      } else if (result.status === 'cancelled') {
        Alert.alert('Cancelled', 'Email sending was cancelled');
      }
    } catch (error) {
      Alert.alert('Error', 'Failed to send email');
      console.error(error);
    }
  };

  // Alternative with HTML formatting
  const sendHtmlEmail = async () => {
    if (!email) {
      Alert.alert('Error', 'Please enter an email address');
      return;
    }

    const available = await checkAvailability();
    if (!available) {
      Alert.alert('Error', 'Email service is not available on this device');
      return;
    }

    const confirmationCode = generateCode();
    setCode(confirmationCode);

    try {
      const result = await MailComposer.composeAsync({
        recipients: [email],
        subject: 'Your Confirmation Code',
        body: `
          <html>
            <body style="font-family: Arial, sans-serif; padding: 20px;">
              <h2 style="color: #333;">Email Confirmation</h2>
              <p>Hello!</p>
              <p>Your confirmation code is:</p>
              <div style="background-color: #f0f0f0; padding: 15px; border-radius: 5px; text-align: center;">
                <h1 style="color: #007AFF; letter-spacing: 5px; margin: 0;">
                  ${confirmationCode}
                </h1>
              </div>
              <p style="color: #666; font-size: 14px;">
                This code will expire in 10 minutes.
              </p>
              <p style="color: #999; font-size: 12px;">
                If you didn't request this code, please ignore this email.
              </p>
            </body>
          </html>
        `,
        isHtml: true,
      });

      if (result.status === 'sent') {
        Alert.alert('Success', 'Confirmation email sent!');
      } else if (result.status === 'cancelled') {
        Alert.alert('Cancelled', 'Email sending was cancelled');
      }
    } catch (error) {
      Alert.alert('Error', 'Failed to send email');
      console.error(error);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Email Confirmation</Text>
      
      <TextInput
        style={styles.input}
        placeholder="Enter email address"
        value={email}
        onChangeText={setEmail}
        keyboardType="email-address"
        autoCapitalize="none"
      />

      <TouchableOpacity style={styles.button} onPress={sendConfirmationEmail}>
        <Text style={styles.buttonText}>Send Plain Text Email</Text>
      </TouchableOpacity>

      <TouchableOpacity style={[styles.button, styles.buttonSecondary]} onPress={sendHtmlEmail}>
        <Text style={styles.buttonText}>Send HTML Email</Text>
      </TouchableOpacity>

      {code ? (
        <View style={styles.codeContainer}>
          <Text style={styles.codeLabel}>Generated Code:</Text>
          <Text style={styles.code}>{code}</Text>
        </View>
      ) : null}
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 20,
    backgroundColor: '#fff',
    justifyContent: 'center',
  },
  title: {
    fontSize: 24,
    fontWeight: 'bold',
    marginBottom: 30,
    textAlign: 'center',
  },
  input: {
    borderWidth: 1,
    borderColor: '#ddd',
    borderRadius: 8,
    padding: 15,
    fontSize: 16,
    marginBottom: 20,
  },
  button: {
    backgroundColor: '#007AFF',
    padding: 15,
    borderRadius: 8,
    marginBottom: 10,
  },
  buttonSecondary: {
    backgroundColor: '#5856D6',
  },
  buttonText: {
    color: '#fff',
    textAlign: 'center',
    fontSize: 16,
    fontWeight: '600',
  },
  codeContainer: {
    marginTop: 30,
    padding: 20,
    backgroundColor: '#f0f0f0',
    borderRadius: 8,
    alignItems: 'center',
  },
  codeLabel: {
    fontSize: 14,
    color: '#666',
    marginBottom: 5,
  },
  code: {
    fontSize: 32,
    fontWeight: 'bold',
    letterSpacing: 5,
    color: '#007AFF',
  },
});
